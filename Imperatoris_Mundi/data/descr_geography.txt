	height
	{
		smooth_ratio_min	0.01			; nominal ratio of noise vs step size for complete smoothness
		smooth_ratio_max	0.25			; nominal ratio of noise vs step size for no smoothness (max sharpness)

		slope_flat			0.2				; maximum gradient of flat regions
		slope_steep			0.5				; nominal gradient of steep regions
		slope_steepest		5.0				; maximum gradient for the terrain

		step_min			0.33			; minimum gradient and noise factor in flat regions
		step_max			2.0				; maximum gradient and noise factor in steep regions
	}

	ground
	{
		slope_flat			0.075			; maximum gradient of flat regions
		slope_factor_flat	0.04			; maximum influence over flat ground
		slope_factor_steep	4.0				; amplify the effect of steep slopes
		curvature_factor	2.0				; amplify the effect of curvature
	}

	river
	{
		valley_factor		0.9				; reduce the river height to a percentage of the sampled height
		valley_bias			-10.0			; adjust the river height by the amount (in metres)

		std_height			150.0			; std height at which the river is the std width
		std_width			64.0			; std width of the river junctions
		tidal_width			192.0			; width at sea level

		ford_width			60.0			; width of the ford (metres)

		noise_trans			0.1				; proportion of segement length deviation during subdivision
		noise_rot			0.1				; standard deviation of rotation applied to direction during subdivision (radians)
		noise_rot_max		0.25			; max rotation (radians)
		noise_width			0.05			; standard deviation of width proportion to apply during subdivision
		noise_width_max		0.25			; max width variation
	}
